\subsection{Poisson--Binomial counting scores}\label{sec:pb}
% Detector gives per-instance objectness s_i; the type head gives per-class
% softmax p_{ik}. Each instance is Bernoulli(w_{ik}), w_{ik}=s_i p_{ik}.
% Count N_k = sum_i Bernoulli(w_{ik}) ~ Poisson--Binomial:
%   E[N_k] = sum_i w_{ik},  Var[N_k] = sum_i w_{ik}(1-w_{ik}),  sigma_k=sqrt(Var+eps).
% K=1 (total count): p collapses to 1, w_i = s_i, still Poisson--Binomial.

\subsection{Joint coverage via a max-statistic}\label{sec:jci}
